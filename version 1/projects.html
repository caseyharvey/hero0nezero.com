function nextBigger(n) {

    let arr = n.toString().split(''),
    high = arr[arr.length - 1],
    head = [],
    tail = [],
    swap,
    res;

    for (i = arr.length - 1; i >= 0; i--) {
        if (arr[i] >= high) {
            high = arr[i];
            tail.unshift(arr[i]);
            arr.pop(arr[i]);
        } else {
            head = [...arr];
            break;
        }
    }

    swap = head[head.length - 1];

    if (tail.length !== n.toString().length) {
        for (j = tail.length - 1; j >= 0; j--) {
            if (tail[j] !== tail[j - 1] && tail[j] > swap || tail[j] > swap) {
                let hold = tail[j];
                tail[j] = swap;
                head[head.length - 1] = hold;
                res = head.concat(tail.sort((a, b) => a - b));
                return parseInt(res.join(''));
            }
        }
    } else {
        return -1;
    }
}
